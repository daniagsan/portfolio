import React, { useState, Children } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
const techCategories = [
  {
    name: 'Frontend',
    items: ['React', 'React Native', 'TypeScript', 'Tailwind CSS']
  },
  {
    name: 'Backend',
    items: ['Python', 'Django', 'Java', 'Node.js']
  },
  {
    name: 'Systems',
    items: ['PostgreSQL', 'Redis', 'Docker', 'Celery']
  },
  {
    name: 'AI',
    items: ['Gemini API', 'Claude API', 'OpenAI API', 'LangChain']
  }];

const containerVariants = {
  hidden: {
    opacity: 0
  },
  visible: {
    opacity: 1,
    transition: {
      staggerChildren: 0.1
    }
  }
};
const itemVariants = {
  hidden: {
    opacity: 0,
    y: 10
  },
  visible: {
    opacity: 1,
    y: 0,
    transition: {
      duration: 0.4
    }
  }
};
function TechCard({ category }: { category: (typeof techCategories)[0]; }) {
  const [isHovered, setIsHovered] = useState(false);
  return (
    <motion.div
      variants={itemVariants}
      onMouseEnter={() => setIsHovered(true)}
      onMouseLeave={() => setIsHovered(false)}
      className="group p-6 rounded-lg border border-gray-200 bg-white hover:bg-black hover:border-black hover:-translate-y-1 transition-all duration-300 shadow-sm hover:shadow-md cursor-default min-h-[180px] flex flex-col">

      <h3 className="font-heading font-semibold text-lg text-gray-900 group-hover:text-white transition-colors duration-300 mb-4">
        {category.name}
      </h3>

      <div className="flex-1 relative">
        <AnimatePresence>
          {isHovered &&
            <motion.div
              initial="hidden"
              animate="visible"
              exit="hidden"
              variants={{
                hidden: {
                  opacity: 0
                },
                visible: {
                  opacity: 1,
                  transition: {
                    staggerChildren: 0.05
                  }
                }
              }}
              className="flex flex-wrap gap-2 absolute inset-0">

              {category.items.map((tech, idx) =>
                <motion.span
                  key={idx}
                  variants={{
                    hidden: {
                      opacity: 0,
                      y: 10
                    },
                    visible: {
                      opacity: 1,
                      y: 0
                    }
                  }}
                  className="font-mono text-xs border border-white/20 rounded-md px-2 py-1 text-white/80 h-fit">

                  {tech}
                </motion.span>
              )}
            </motion.div>
          }
        </AnimatePresence>
      </div>
    </motion.div>);

}
export function TechStackSection() {
  return (
    <section className="w-full" id="stack">
      <div className="mb-12">
        <span className="font-mono text-xs text-gray-400 mb-2 block">
          {' '}
          // tools
        </span>
        <h2 className="font-heading text-2xl font-semibold text-black">
          Technology Stack
        </h2>
      </div>

      <motion.div
        variants={containerVariants}
        initial="hidden"
        whileInView="visible"
        viewport={{
          once: true,
          margin: '-100px'
        }}
        className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-6">

        {techCategories.map((category, idx) =>
